EXTENSIONES NECESARIAS CREATE EXTENSION IF NOT EXISTS "uuid-ossp"; -- 2. CREACIÓN DE TABLAS -- Perfiles y Balances (Moneda Ficticia) CREATE TABLE IF NOT EXISTS public.profiles ( id UUID REFERENCES auth.users NOT NULL PRIMARY KEY, username TEXT UNIQUE, full_name TEXT, avatar_url TEXT, balance NUMERIC(15,2) DEFAULT 0.00 CHECK (balance >= 0), vip_level TEXT DEFAULT 'Bronce', is_approved BOOLEAN DEFAULT FALSE, invite_code TEXT UNIQUE, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Invitaciones (Los 12 Elegidos) CREATE TABLE IF NOT EXISTS public.invitations ( id UUID DEFAULT uuid_generate_v4() PRIMARY KEY, code TEXT UNIQUE NOT NULL, is_used BOOLEAN DEFAULT FALSE, created_by UUID REFERENCES auth.users, used_by UUID REFERENCES auth.users, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Historial de Apuestas CREATE TABLE IF NOT EXISTS public.bets ( id UUID DEFAULT uuid_generate_v4() PRIMARY KEY, user_id UUID REFERENCES public.profiles(id) NOT NULL, game_type TEXT NOT NULL, -- 'roulette', 'blackjack', 'poker', 'slots' game_id TEXT NOT NULL, bet_amount NUMERIC(15,2) NOT NULL, payout_amount NUMERIC(15,2) DEFAULT 0.00, status TEXT DEFAULT 'pending', -- 'pending', 'completed', 'cancelled' bet_selection JSONB, created_at TIMESTAMPTZ DEFAULT NOW() ); -- Transacciones (PayPal / Binance) CREATE TABLE IF NOT EXISTS public.transactions ( id UUID DEFAULT uuid_generate_v4() PRIMARY KEY, user_id UUID REFERENCES public.profiles(id) NOT NULL, amount NUMERIC(15,2) NOT NULL, type TEXT NOT NULL, -- 'deposit', 'withdrawal' method TEXT NOT NULL, -- 'paypal', 'binance' status TEXT DEFAULT 'pending', -- 'pending', 'approved', 'rejected' description TEXT, created_at TIMESTAMPTZ DEFAULT NOW() ); -- 3. FUNCIONES DE LÓGICA SEGURA (RPC) -- Procesamiento Seguro de Apuestas (Descuenta balance y registra) CREATE OR REPLACE FUNCTION public.process_bet_secure( p_user_id UUID, p_amount NUMERIC, p_game_type TEXT, p_game_id TEXT, p_selection JSONB ) RETURNS UUID AS $$ DECLARE v_bet_id UUID; BEGIN -- Verificar saldo suficiente IF (SELECT balance FROM public.profiles WHERE id = p_user_id) < p_amount THEN RAISE EXCEPTION 'Saldo insuficiente para realizar la apuesta'; END IF; -- Descontar del balance UPDATE public.profiles SET balance = balance - p_amount WHERE id = p_user_id; -- Registrar apuesta INSERT INTO public.bets (user_id, game_type, game_id, bet_amount, bet_selection) VALUES (p_user_id, p_game_type, p_game_id, p_selection) RETURNING id INTO v_bet_id; RETURN v_bet_id; END; $$ LANGUAGE plpgsql SECURITY DEFINER; -- 4. SEGURIDAD DE FILAS (RLS) ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY; CREATE POLICY "Usuarios pueden ver su propio perfil" ON public.profiles FOR SELECT USING (auth.uid() = id); ALTER TABLE public.bets ENABLE ROW LEVEL SECURITY; CREATE POLICY "Usuarios pueden ver sus propias apuestas" ON public.bets FOR SELECT USING (auth.uid() = user_id); ALTER TABLE public.transactions ENABLE ROW LEVEL SECURITY; CREATE POLICY "Usuarios pueden ver sus transacciones" ON public.transactions FOR SELECT USING (auth.uid() = user_id); -- 5.
